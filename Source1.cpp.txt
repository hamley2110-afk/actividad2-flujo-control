#include <iostream>
#include <iomanip>

using namespace std;

int main() {
    double sueldoMensual, sueldoAnual, isr = 0;

    cout << "Ingrese el sueldo mensual del empleado: RD$ ";
    cin >> sueldoMensual;

    sueldoAnual = sueldoMensual * 12;

    // Cálculo del ISR anual (RD - año actual)
    if (sueldoAnual <= 416220.00) {
        isr = 0;
    }
    else if (sueldoAnual <= 624329.00) {
        isr = (sueldoAnual - 416220.01) * 0.15;
    }
    else if (sueldoAnual <= 867123.00) {
        isr = 31216.00 + (sueldoAnual - 624329.01) * 0.20;
    }
    else {
        isr = 79776.00 + (sueldoAnual - 867123.01) * 0.25;
    }

    cout << fixed << setprecision(2);
    cout << "\n========== RESULTADOS ==========" << endl;
    cout << "Sueldo mensual: RD$ " << sueldoMensual << endl;
    cout << "Sueldo anual:   RD$ " << sueldoAnual << endl;

    if (isr == 0)
        cout << "ISR a descontar: RD$ 0.00" << endl;
    else
        cout << "ISR anual: RD$ " << isr << endl;

    return 0;
}
